import { palette } from '@guardian/source-foundations';
import { Meta } from '@storybook/addon-docs';

<Meta
	title="Styles/Colour"
	parameters={{ previewTabs: { canvas: { hidden: true } } }}

# Colours

<div className="colours">
	{Object.keys(palette).map((colourCategory) => {
		const colours = Object.keys(palette[colourCategory]).map((colour) => {
			return (
				<div className="colours__row">
					<div className="colours__value">
						<div
							className="colours__value__swatch"
							style={{
								'background-color':
									palette[colourCategory][colour],
							}}
						/>
						<span>{palette[colourCategory][colour]}</span>
					</div>
					<span>
						{colourCategory}.{colour}
					</span>
				</div>
			);
		});
		return <>{colours}</>;
	})}
</div>
